```python
x = ["apple", "banana", "cherry", "durian"]
print(x[0]) # will print apple
x.append("eggplant") # will add "eggplant" to the list
x[3] = "cake" # will change "cherry" to "cake"
print(x) # will print the whole list
print(9 in x) # will print False
# print all the items in a list
for i in range(len(x)):
	print(x[i])
# print all the items in a list without index
for item in x:
	print(item)

```
